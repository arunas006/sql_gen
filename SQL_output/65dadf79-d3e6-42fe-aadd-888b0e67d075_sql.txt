SELECT
  c.state,
  SUM(s.total_sales_amount) AS total_sales
FROM
  sales s
JOIN
  customer c ON s.customer_id = c.customer_id
WHERE
  s.sales_date >= date_trunc('quarter', CURRENT_DATE) - INTERVAL '3 months'
  AND s.sales_date < date_trunc('quarter', CURRENT_DATE)
GROUP BY
  c.state
ORDER BY
  total_sales DESC;
